#[derive(Default)]
struct ListNode<T> {
    data: T,
    next: *mut ListNode<T>,
    prev: *mut ListNode<T>,
}

impl<T> ListNode<T> {
    fn new(value: T) -> *mut ListNode<T> {
        Box::into_raw(Box::new(ListNode {
            data: value,
            next: ptr::null_mut(),
            prev: ptr::null_mut(),
        }))
    }
}

#[derive(Default)]
pub struct LinkedList<T> {
    head: *mut ListNode<T>,
    tail: *mut ListNode<T>,
    len: usize,
}
